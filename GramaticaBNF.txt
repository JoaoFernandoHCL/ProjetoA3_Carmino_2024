Gramática - BNF


<declaracao> ::= <atribuicao>  
                 | <expressao> ";"
                 | <declaracao_se> 
                 | <declaracao_enquanto> 
                 | <print> 
 
<atribuicao> ::=  <nome> “=” <expressao> ”;” 

<declaracao_se> ::= ”se” ”(“ <condicao> ”)” ”{” <declaracao> ”}”["senao" <declaracao>]
 
<declaracao_enquanto> ::= "enquanto" ”(“ <condicao> “)” “{“ <declaracao> “}” 

<print> ::= "print" “(“<expressao> | <string>”)” ";" 

<condicao> ::= <termoBool> { ("||") <termoBool> }

<termoBool> ::= <fatorBool> { ("&&") <fatorBool> } 

<fatorBool> ::= <expressao> ”<” | ”<=” | ”>” | ”>=” | ”==” | ”!=” <expressao> 

<expressao> ::= <termo> { ("+" | "-") <termo> }

<termo> ::= <fator> { ("*" | "/") <fator> } 

<fator> ::= <numero>  
            | <nome>  
            | <string> 
            | "(" <expr> ")" 

<nome> ::= [a-zA-Z_][a-zA-Z0-9_]* 

<string> ::= ‘”’ .* ‘”’  

<numero> ::= [0-9]+
